Require Import Basics Types Pointed WildCat.
Require Import AbGroups.AbelianGroup AbSES.Core.

(** * The set [Ext B A] of abelian group extensions *)

Definition Ext (B A : AbGroup) := Tr 0 (AbSES B A).

Instance ispointed_ext {B A : AbGroup} : IsPointed (Ext B A) := tr (point _).

(** An extension [E : AbSES B A] is trivial in [Ext B A] if and only if [E] merely splits. *)

H:Univalence
B, A:AbGroup
E:AbSES B A
merely
  {s : GroupHomomorphism B E &
  projection E $o s == idmap} <~>
tr E = point (Ext B A)


H:Univalence
B, A:AbGroup
E:AbSES B A
merely
  {s : GroupHomomorphism B E &
  projection E $o s == idmap} <~>
tr E = point (Ext B A)

  
H:Univalence
B, A:AbGroup
E:AbSES B A
merely
  {s : GroupHomomorphism B E &
  projection E $o s == idmap} <~>
Tr (-1) (E = point (AbSES B A))

  srapply equiv_iff_hprop;
    apply Trunc_functor;
    apply iff_abses_trivial_split.
Defined.
